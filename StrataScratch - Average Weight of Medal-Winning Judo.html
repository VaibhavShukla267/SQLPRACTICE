-- min(if(age=20,age,null)) as min_age
-- group by team,average_weigth
SELECT team, AVG(weight) AS average_weight
FROM olympics_athletes_events
WHERE medal='Gold'or medal='Silver'or medal='Bronze' and age>19 or age<31
GROUP BY team
having average_weight is not null;
